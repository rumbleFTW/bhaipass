a. DematAccount.java
package Account;
public class DematAccount
{
 private int id;
 String name;
 String address;
 int panNumber;
 public DematAccount(int id, String name, String address, int panNumber)
 {
 this.id = id;
 this.name = name;
 this.address = address;
 this.panNumber = panNumber;
 }
 public void update(String name, String address, int panNumber)
 {
 this.name = name;
 this.address = address;
 this.panNumber = panNumber;
 }
 public String toString()
 {
 return("ID: "+this.id+", Name: "+this.name+", Address: "+this.address+", PAN Number:
"+this.panNumber+", Account type: Demat");
 }
}
